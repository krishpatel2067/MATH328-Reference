\section{Vector Analysis}

\subsection{Scalar Fields}

A scalar field $f(\mathbf{x})$ is a function that maps each point $\mathbf{x} = x_1\hat{x}_1 + x_2\hat{x}_2 + \cdots + x_n\hat{x}_n$ to a scalar: $f: \mathbb{R}^n \to \mathbb{R}$. Some examples include temperature and pressure as a function of position.

\subsection{Vector Fields}

A vector field $\mathbf{F}(\mathbf{x}) = F_1(\mathbf{x})\hat{x}_1 + F_2(\mathbf{x})\hat{x}_2 + \cdots + F_m(\mathbf{x})\hat{x}_m$ is a function that maps each point $\mathbf{x} = x_1\hat{x}_1 + x_2\hat{x}_2 + \cdots + x_n\hat{x}_n$ to a vector: $\mathbf{F}: \mathbb{R}^n \to \mathbb{R}^m$. Note that each $F_i(\mathbf{x})$ is a scalar field. Some examples include velocity and acceleration as a function of position.

\subsection{Dot Product}

Also known as the scalar product, the dot product takes two vectors and returns a scalar.

\begin{align*}
    \mathbf{A} \cdot \mathbf{B} & = |\mathbf{A}|\mathbf{B}|\cos\theta \\
                                & = A_x B_x + A_y B_y + A_z B_z
\end{align*}

Properties:
\begin{itemize}
    \item Commutative: $\mathbf{A}\cdot\mathbf{B} = \mathbf{B}\cdot\mathbf{A}$
    \item Distributive: $\mathbf{A}\cdot(\mathbf{B}+\mathbf{C}) = \mathbf{A}\cdot\mathbf{B} + \mathbf{A}\cdot\mathbf{C}$
    \item $\mathbf{A}\cdot\mathbf{A} = |\mathbf{A}|^2$
    \item $\mathbf{A}\cdot\mathbf{B} = 0 \iff \mathbf{A} \perp \mathbf{B}$
\end{itemize}

Note that the dot product is \emph{not} associative.

\subsection{Projections}
The scalar projection of $\mathbf{A}$ onto $\mathbf{B}$ is given by

\begin{equation*}
    \frac{\mathbf{A}\cdot\mathbf{B}}{|\mathbf{B}|} = |\mathbf{A}|\cos\theta
\end{equation*}

The vector projection of $\mathbf{A}$ onto $\mathbf{B}$ is given by

\begin{equation*}
    \frac{\mathbf{A}\cdot\mathbf{B}}{|\mathbf{B}|^2}\mathbf{B} = (\mathbf{A}\cdot\mathbf{u})\mathbf{u}
\end{equation*}

\subsection{Gram-Schmidt Orthonormalization}

Given any two vectors $\mathbf{A}$ and $\mathbf{B}$, we can find an orthonormal basis $\mathbf{v_1}$ and $\mathbf{v_2}$ for the plane spanned by $\mathbf{A}$ and $\mathbf{B}$ as
\begin{align*}
    \mathbf{v_1} & = \frac{\mathbf{A}}{|\mathbf{A}|} = \mathbf{u}                                                                    \\
    \mathbf{v_2} & = \frac{\mathbf{B} - (\mathbf{B}\cdot\mathbf{u})\mathbf{u}}{|\mathbf{B} - (\mathbf{B}\cdot\mathbf{u})\mathbf{u}|}
\end{align*}

\subsection{Equation of a Plane}

\begin{align*}
    \mathbf{n} \cdot (\mathbf{r} - \mathbf{r_0}) & = 0 \\
    n_x(x-x_0) + n_y(y-y_0) + n_z(z-z_0)         & = 0
\end{align*}

where
\begin{itemize}
    \item $\mathbf{r} = (x, y, z)$ is a vector to any point on the plane,
    \item $\mathbf{n}$ is the normal vector to the plane, and
    \item $\mathbf{r_0} = (x_0, y_0, z_0)$ is a known point on the plane.
\end{itemize}

\subsection{Cross Product}

Also known as the vector product, the cross product takes two vectors and returns a vector.

\begin{align*}
    \mathbf{A} \times \mathbf{B} & = (|\mathbf{A}||\mathbf{B}|\sin\theta)\mathbf{u} \\
                                 & = \begin{vmatrix}
                                         \mathbf{i} & \mathbf{j} & \mathbf{k} \\
                                         A_x        & A_y        & A_z        \\
                                         B_x        & B_y        & B_z
                                     \end{vmatrix}
\end{align*}

where $\mathbf{u}$ is the unit vector perpendicular to both $\mathbf{A}$ and $\mathbf{B}$. Use the \textbf{right-hand rule} to determine the correct direction of $\mathbf{u}$. Geometrically, the magnitude of the cross product represents the area of the parallelogram created by $\mathbf{A}$ and $\mathbf{B}$.

\vsp

Properties:
\begin{itemize}
    \item Anti-commutative: $\mathbf{A} \times \mathbf{B} = -\mathbf{B} \times \mathbf{A}$
    \item Distributive: $\mathbf{A} \times (\mathbf{B} + \mathbf{C}) = \mathbf{A} \times \mathbf{B} + \mathbf{A} \times \mathbf{C}$
    \item $\mathbf{A} \times \mathbf{B} = 0 \iff \mathbf{A} \parallel \mathbf{B}$
          \begin{itemize}
              \item Thus, $\mathbf{A} \times \mathbf{A} = 0$.
          \end{itemize}
\end{itemize}

\subsection{Equation of a 3D Line}

\begin{align*}
    \mathbf{r}                   & = \mathbf{r}_0 + t\mathbf{u}
    \\
    \mathbf{r} \times \mathbf{u} & = \mathbf{r}_0 \times \mathbf{u}
\end{align*}

where
\begin{itemize}
    \item $\mathbf{r} = (x, y, z)$ is a vector to any point on the line,
    \item $\mathbf{r}_0 = (x_0, y_0, z_0)$ is a known point on the line,
    \item $\mathbf{u}$ is a unit vector in the direction of the line, and
    \item $t$ is a scalar parameter.
\end{itemize}

\subsection{Triple Scalar Product}
\begin{equation*}
    \mathbf{A}\cdot(\mathbf{B} \times \mathbf{C}) =
    \left|\begin{matrix}
        A_x & A_y & A_z \\
        B_x & B_y & B_z \\
        C_x & C_y & C_z
    \end{matrix}\right|
\end{equation*}

Performing an even number of swaps on the determinant results in the same answer (which is equivalent to performing \textbf{cyclic permutations} such as $\mathbf{A}\cdot(\mathbf{B}\times\mathbf{C}) = \mathbf{B}\cdot(\mathbf{C}\times\mathbf{A}) = \mathbf{C}\cdot(\mathbf{A}\times\mathbf{B})$), whereas performing an odd number of swaps simply results in an answer of the opposite sign.

Geometrically, the triple scalar product represents the volume of a parallelopiped formed by $\mathbf{A}$, $\mathbf{B}$, and $\mathbf{C}$. More specifically, any two vectors can be used to form the base of the parallelopiped (e.g., $\mathbf{A}$ and $\mathbf{B}$), and the height is the third vector's ($\mathbf{C}$) projection onto the normal vector of the base ($\mathbf{A} \times \mathbf{B}$). Furthermore, $\frac{1}{6}$ of the volume of the parallelopiped is the volume of the tetrahedron formed by $\mathbf{A}$, $\mathbf{B}$, and $\mathbf{C}$.
\subsection{Triple Vector Product}
\begin{equation*}
    \mathbf{A} \times (\mathbf{B} \times \mathbf{C})=\mathbf{B}(\mathbf{A} \cdot \mathbf{C}) - \mathbf{C}(\mathbf{A} \cdot \mathbf{B})
\end{equation*}

To remember this formula (beyond the mnemonic BAC-CAB), the right hand side is a linear combination of the vectors in the parentheses on the left hand side. The coefficients are the outside vector dotted with each of the inside vectors. Both terms have $\mathbf{A}$, $\mathbf{B}$, and $\mathbf{C}$ exactly once. Finally, the positive term on the right hand side belongs to the vector in the middle on the left hand side ($\mathbf{B}$ in this case).

$\mathbf{B} \times \mathbf{C}$ is perpendicular to the plane spanned by $\mathbf{B}$ and $\mathbf{C}$, but $\mathbf{A} \times (\mathbf{B} \times \mathbf{C})$ is back on that plane.

\subsection{Gradient}

The gradient represents the direction and magnitude of the greatest rate of increase of a scalar field. The vector opposite of the gradient is the direction of the greatest rate of decrease. The vectors perpendicular to the gradient represent no change in the scalar field.

\vsp

In rectangular coordinates:
\begin{equation*}
    \bm{\nabla} f = \text{grad}~f = \frac{\partial f}{\partial x}\mathbf{i} + \frac{\partial f}{\partial y}\mathbf{j} + \frac{\partial f}{\partial z}\mathbf{k}
\end{equation*}

where $f$ is a scalar field. Note that $\bm{\nabla}f$
\begin{itemize}
    \item is a vector field.
    \item points in the direction of the greatest rate of increase of $f$.
    \item is perpendicular to the surface $f = c$ for any constant $c$.
\end{itemize}

In cylindrical coordinates:
\begin{equation*}
    \bm{\nabla} f = \text{grad}~f = \frac{\partial f}{\partial r}\mathbf{e}_r + \frac{1}{r}\frac{\partial f}{\partial \theta}\mathbf{e}_\theta + \frac{\partial f}{\partial z}\mathbf{e}_z
\end{equation*}

In spherical coordinates:
\begin{equation*}
    \bm{\nabla} f = \text{grad}~f = \frac{\partial f}{\partial r}\mathbf{e}_r + \frac{1}{r}\frac{\partial f}{\partial \theta}\mathbf{e}_\theta + \frac{1}{r\sin\theta}\frac{\partial f}{\partial \phi}\mathbf{e}_\phi
\end{equation*}

\subsection{Directional Derivative}

\begin{equation*}
    \frac{d\phi}{ds} = \bm{\nabla} \phi \cdot \mathbf{u}
\end{equation*}

where $\mathbf{u}$ is the unit vector in the direction of differentiation.

Special cases:
\begin{itemize}
    \item $\mathbf{u} = \mathbf{i} \implies \frac{d\phi}{ds} = \frac{\partial \phi}{\partial x}$.
    \item $\mathbf{u} = \mathbf{j} \implies \frac{d\phi}{ds} = \frac{\partial \phi}{\partial y}$.
    \item $\mathbf{u} = \mathbf{k} \implies \frac{d\phi}{ds} = \frac{\partial \phi}{\partial z}$.
\end{itemize}

\subsection{Divergence}

In rectangular coordinates:
\begin{equation*}
    \bm{\nabla} \cdot \mathbf{V} = \text{div}~\mathbf{V} = \frac{\partial V_x}{\partial x} + \frac{\partial V_y}{\partial y} + \frac{\partial V_z}{\partial z}
\end{equation*}

where $\mathbf{V}$ is a vector field. Note that $\bm{\nabla}\cdot\mathbf{V}$ is a scalar field.

In cylindrical coordinates:
\begin{equation*}
    \bm{\nabla} \cdot \mathbf{V} = \frac{1}{r}\frac{\partial}{\partial r}(rV_r) + \frac{1}{r}\frac{\partial}{\partial \theta}V_\theta + \frac{\partial}{\partial z}V_z
\end{equation*}

In spherical coordinates:
\begin{equation*}
    \bm{\nabla} \cdot \mathbf{V} = \frac{1}{r^2}\frac{\partial}{\partial r}(r^2V_r) + \frac{1}{r\sin\theta}\frac{\partial}{\partial \theta}(V_\theta \sin\theta) + \frac{1}{r\sin\theta}\frac{\partial V_\phi}{\partial \phi}
\end{equation*}

\subsection{Curl}

In rectangular coordinates:
\begin{align*}
    \bm{\nabla} \times \mathbf{V} & = \text{curl}~\mathbf{V}                                                                                                                                                                                                                                                                  \\
                                  & = \begin{vmatrix}
                                          \mathbf{i}                  & \mathbf{j}                  & \mathbf{k}                  \\
                                          \frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
                                          V_x                         & V_y                         & V_z
                                      \end{vmatrix}                                                                                                                                                                                                 \\
                                  & = \left( \frac{\partial V_z}{\partial y} - \frac{\partial V_y}{\partial z} \right) \mathbf{i} + \left( \frac{\partial V_x}{\partial z} - \frac{\partial V_z}{\partial x} \right) \mathbf{j} + \left( \frac{\partial V_y}{\partial x} - \frac{\partial V_x}{\partial y} \right) \mathbf{k}
\end{align*}

where $\mathbf{V}$ is a vector field. Note that $\bm{\nabla}\times\mathbf{V}$ is a vector field.

\subsection{Laplacian}

In rectangular coordinates:
\begin{align*}
    \nabla^2 f = \bm{\nabla}\cdot(\bm{\nabla}f) & = \text{div}~\text{grad}~f                                                                                  \\
                                                & = \frac{\partial^2 f}{\partial x^2} + \frac{\partial^2 f}{\partial y^2} + \frac{\partial^2 f}{\partial z^2}
\end{align*}

In cylindrical coordinates:
\begin{equation*}
    \nabla^2 f = \frac{1}{r}\frac{\partial}{\partial r}\left(r\frac{\partial f}{\partial r}\right) + \frac{1}{r^2}\frac{\partial^2 f}{\partial \theta^2} + \frac{\partial^2 f}{\partial z^2}
\end{equation*}

In spherical coordinates:
\begin{equation*}
    \nabla^2 f = \frac{1}{r^2}\frac{\partial}{\partial r}\left(r^2\frac{\partial f}{\partial r}\right) + \frac{1}{r^2\sin\theta}\frac{\partial}{\partial \theta}\left(\sin\theta\frac{\partial f}{\partial \theta}\right) + \frac{1}{r^2\sin^2\theta}\frac{\partial^2 f}{\partial \phi^2}
\end{equation*}

\subsection{Vector Operator Properties}

\begin{itemize}
    \item A field is \textbf{irrotational} iff $\bm{\nabla} \times \mathbf{F} = \mathbf{0}$
    \item A field is \textbf{incompressible} (or \textbf{solenoidal}) iff $\bm{\nabla} \cdot \mathbf{F} = 0$
          \begin{itemize}
              \item $\bm{\nabla} \cdot (\bm{\nabla} \times \mathbf{F}) = 0$
          \end{itemize}
    \item Linearity:
          \begin{itemize}
              \item $\bm{\nabla}(f + g) = \bm{\nabla}f + \bm{\nabla}g$ and $\bm{\nabla}(cf) = c\bm{\nabla}f$
              \item $\bm{\nabla}\cdot(\mathbf{F} + \mathbf{G}) = \bm{\nabla}\cdot\mathbf{F} + \bm{\nabla}\cdot\mathbf{G}$ and $\bm{\nabla}\cdot(c\mathbf{F}) = c(\bm{\nabla}\cdot\mathbf{F})$
              \item $\bm{\nabla}\times(\mathbf{F} + \mathbf{G}) = \bm{\nabla}\times\mathbf{F} + \bm{\nabla}\times\mathbf{G}$ and $\bm{\nabla}\times(c\mathbf{F}) = c(\bm{\nabla}\times\mathbf{F})$
          \end{itemize}
    \item Product rule:
          \begin{itemize}
              \item $\bm{\nabla}(fg) = f\bm{\nabla}g + g\bm{\nabla}f$
              \item $\bm{\nabla} \cdot (f\mathbf{F}) = f(\bm{\nabla} \cdot \mathbf{F}) + \mathbf{F} \cdot (\bm{\nabla}f)$
              \item $\bm{\nabla} \times (f\mathbf{F}) = f(\bm{\nabla} \times \mathbf{F}) + (\bm{\nabla}f) \times \mathbf{F}$
          \end{itemize}
    \item Gradient quotient rule:
          \begin{equation*}
              \bm{\nabla}\left(\frac{f}{g}\right) = \frac{g\bm{\nabla}f - f\bm{\nabla}g}{g^2}
          \end{equation*}
    \item $\bm{\nabla} \cdot (\mathbf{F} \times \mathbf{G}) = \mathbf{G} \cdot (\bm{\nabla} \times \mathbf{F}) - \mathbf{F} \cdot (\bm{\nabla} \times \mathbf{G})$
    \item $\nabla^2(fg) = f~\nabla^2 g + g~\nabla^2 f + 2(\bm{\nabla}f \cdot \bm{\nabla}g)$
    \item Using properties above: $\bm{\nabla} \cdot (\bm{\nabla} f \times \bm{\nabla} g) = 0$
    \item $\bm{\nabla} \cdot (f \bm{\nabla} g - g \bm{\nabla} f) = f~\nabla^2 g - g~\nabla^2 f$
    \item Using the triple vector product: $\bm{\nabla} \times (\bm{\nabla} \times \mathbf{F}) = \bm{\nabla}(\bm{\nabla} \cdot \mathbf{F}) - \nabla^2 \mathbf{F}$
          \begin{itemize}
              \item Note that this property also defines $\nabla^2 \mathbf{F}$.
          \end{itemize}
\end{itemize}
